문제

현재 미세먼지 농도를 입력하면 그 결과를 바탕으로 좋음, 보통, 나쁨, 매우나쁨을 출력하는 문제입니다 이를 해결하는 다음 프로그램의 실행상태를 확인하세요 참고로, 현재 미세먼지 농도( ㎍/m³)가 30이하이면 '좋음', 30 - 80이면 '보통', 80 - 150이면 '나쁨', 151이상이면 '매우나쁨'입니다.
0 ~ 200 사이에서 현재 미세먼지 농도를 입력하세요 : 60
미세먼지 농도는 대부분의 지역이 '보통' 수준을 보이고 있다

프로그램 코드

	#include <stdio.h> 

1	int main()
	{
2		int fineDust = 0;

3		printf("0 ~ 200 사이에서 현재 미세먼지 농도를 입력하세요 : " );
4		scanf( "%d", &fineDust );
	

5		if ( fineDust <= 30 )								
		{
6			printf( "미세먼지 농도는 대부분의 지역이 '좋음' 수준을 보이고 있다 " );
		}
7		else if ( ( 30 < fineDust ) && ( fineDust <= 80 ) )				
		{
8			printf( "미세먼지 농도는 대부분의 지역이 '보통' 수준을 보이고 있다 " );
		}
9		else if ( ( 80 < fineDust ) && ( fineDust <= 150 ) )				
		{
10			printf( "미세먼지 농도는 대부분의 지역이 '나쁨' 수준을 보이고 있다 " );
		}
11		else 										
		{
12			printf( "미세먼지 농도는 대부분의 지역이 '매우나쁨' 수준을 보이고 있다 " );
		}
		
13		return 0;
	}
