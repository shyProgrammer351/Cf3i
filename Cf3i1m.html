문제

변수에 값을 저장할 때 발생하는 overflow와 underflow를 실습하는 문제입니다 이를 해결하는 다음 프로그램의 실행상태를 확인하세요 
max     =  2147483647
max + 1 = -2147483648 ( overflow발생 ) 
min     = -2147483648
min - 1 =  2147483647 ( underflow발생 )

프로그램 코드

	#include <stdio.h>
	
1	int main() {	
		int max = +2147483647;			
2		int min = -2147483648;			
	
3		int overflow = max + 1;			
	
4		int underflow = min - 1;	 
	
5		printf( "max     =  %d\n", max );
6		printf( "max + 1 = %d ( overflow 발생 )\n", overflow );	 
7		printf( "min     = %d\n", min );
8		printf( "min - 1 =  %d ( underflow 발생 )\n", underflow ); 
	
9		return 0;
	}

1 2와 5 6 7 사이의 실행순서(2개) 및 실행상태

3  int overflow = max + 1;
main() 함수
+2147483647 max overflow
-2147483648 min 0 underflow
 

4  int underflow = min - 1;
main() 함수
+2147483647 max -2147483648 overflow
-2147483648 min underflow
